77      @Override
78      public JsonDeserializer<?> createContextual(DeserializationContext ctxt,
79              BeanProperty property) throws JsonMappingException
80      {
81          JsonFormat.Value format = findFormatOverrides(ctxt, property, handledType());
82          ThreeTenDateTimeDeserializerBase<?> deser = this;
83          if (format != null) {
84              if (format.hasPattern()) {
85                  final String pattern = format.getPattern();
86                  final Locale locale = format.hasLocale() ? format.getLocale() : ctxt.getLocale();
87                  DateTimeFormatterBuilder builder = new DateTimeFormatterBuilder();
88                  if (acceptCaseInsensitiveValues(ctxt, format)) {
89                      builder.parseCaseInsensitive();
90                  }
91                  builder.appendPattern(pattern);
92                  DateTimeFormatter df;
93                  if (locale == null) {
94                      df = builder.toFormatter();
95                  } else {
96                      df = builder.toFormatter(locale);
97                  }
98                  //Issue #69: For instant serializers/deserializers we need to configure the formatter with
99                  //a time zone picked up from JsonFormat annotation, otherwise serialization might not work
100                 if (format.hasTimeZone()) {
101                     df = df.withZone(DateTimeUtils.toZoneId(format.getTimeZone()));
102                 }
103                 deser = deser.withDateFormat(df);
104             }
105             // 17-Aug-2019, tatu: For 2.10 let's start considering leniency/strictness too
106             if (format.hasLenient()) {
107                 Boolean leniency = format.getLenient();
108                 if (leniency != null) {
109                     deser = deser.withLeniency(leniency);
110                 }
111             }
112             // any use for TimeZone?
113         }
114         return deser;
115     }

126     private boolean acceptCaseInsensitiveValues(DeserializationContext ctxt, JsonFormat.Value format) 
127     {
128         Boolean enabled = format.getFeature( Feature.ACCEPT_CASE_INSENSITIVE_VALUES);
129         if( enabled == null) {
