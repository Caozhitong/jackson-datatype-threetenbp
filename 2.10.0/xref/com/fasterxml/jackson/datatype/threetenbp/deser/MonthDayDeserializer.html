37      @Override
38      public MonthDay deserialize(JsonParser parser, DeserializationContext context) throws IOException
39      {
40          if (parser.hasToken(JsonToken.VALUE_STRING)) {
41              String string = parser.getValueAsString().trim();
42              try {
43                  if (_formatter == null) {
44                      return MonthDay.parse(string);
45                  }
46                  return MonthDay.parse(string, _formatter);
47              } catch (DateTimeException e) {
48                  return _handleDateTimeException(context, e, string);
49              }
50          }
51          if (parser.isExpectedStartArrayToken()) {
52              JsonToken t = parser.nextToken();
53              if (t == JsonToken.END_ARRAY) {
54                  return null;
55              }
56              if ((t == JsonToken.VALUE_STRING || t == JsonToken.VALUE_EMBEDDED_OBJECT)
57                      && context.isEnabled(DeserializationFeature.UNWRAP_SINGLE_VALUE_ARRAYS)) {
58                  final MonthDay parsed = deserialize(parser, context);
59                  if (parser.nextToken() != JsonToken.END_ARRAY) {
60                      handleMissingEndArrayForSingle(parser, context);
61                  }
62                  return parsed;
63              }
64              if (t != JsonToken.VALUE_NUMBER_INT) {
65                  _reportWrongToken(context, JsonToken.VALUE_NUMBER_INT, "month");
66              }
67              int month = parser.getIntValue();
68              int day = parser.nextIntValue(-1);
69              if (day == -1) {
70                  if (!parser.hasToken(JsonToken.VALUE_NUMBER_INT)) {
71                      _reportWrongToken(context, JsonToken.VALUE_NUMBER_INT, "day");
72                  }
73                  day = parser.getIntValue();
74              }
75              if (parser.nextToken() != JsonToken.END_ARRAY) {
76                  throw context.wrongTokenException(parser, handledType(), JsonToken.END_ARRAY,
77                          "Expected array to end");
78              }
79              return MonthDay.of(month, day);
80          }
81          if (parser.hasToken(JsonToken.VALUE_EMBEDDED_OBJECT)) {
82              return (MonthDay) parser.getEmbeddedObject();
83          }
84          return _handleUnexpectedToken(context, parser,
85                  JsonToken.VALUE_STRING, JsonToken.START_ARRAY);
86      }
